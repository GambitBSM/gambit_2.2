68 {
69 const str be = "Pythia" + model_suffix;
70 const str ver = Backends::backendInfo().default_version(be);
71 pythia_doc_path = Backends::backendInfo().path_dir(be, ver) + "/../share/Pythia8/xmldoc/";
72 result.banner(pythia_doc_path);
73 first = false;

77 // To make sure that the Pythia instance on each OMP thread gets all the
78 // options it should, all the options parsing and initialisation happens in
79 // COLLIDER_INIT_OMP (OMP parallel) rather than COLLIDER_INIT (only thread 0).
80 // We may want to split this up, so that all the yaml options are parsed in
81 // COLLIDER_INIT (by thread 0), and used to initialize the 'result' instance
82 // of each thread within COLLIDER_INIT_OMP.
83 //
84 // else if (iteration == COLLIDER_INIT)

86 // // Do the option parsing here?

92 std::vector<str> pythiaOptions;
93
94 // By default we tell Pythia to be quiet. (Can be overridden from yaml settings)
95 pythiaOptions.push_back("Print:quiet = on");
96 pythiaOptions.push_back("SLHA:verbose = 0");
97
98 // Get options from yaml file.
99 const double xsec_veto_default = 0.0;
100 const bool partonOnly_default = false;
101 const double antiktR_default = 0.4;
102 if (runOptions.hasKey(RunMC.current_collider()))
103 {
104 YAML::Node colNode = runOptions.getValue<YAML::Node>(RunMC.current_collider());
105 Options colOptions(colNode);
106 xsec_veto_fb = colOptions.getValueOrDef<double>(xsec_veto_default, "xsec_veto");
107 result.partonOnly = colOptions.getValueOrDef<bool>(partonOnly_default, "partonOnly");
108 result.antiktR = colOptions.getValueOrDef<double>(antiktR_default, "antiktR");
109 if (colOptions.hasKey("pythia_settings"))
110 {
111 std::vector<str> addPythiaOptions = colNode["pythia_settings"].as<std::vector<str> >();
112 pythiaOptions.insert(pythiaOptions.end(), addPythiaOptions.begin(), addPythiaOptions.end());
113 }
114 }
115 else
116 {
117 xsec_veto_fb = xsec_veto_default;
118 result.partonOnly = partonOnly_default;
119 result.antiktR = antiktR_default;
120 }
121
122 // We need showProcesses for the xsec veto.
123 pythiaOptions.push_back("Init:showProcesses = on");
124
125 // We need "SLHA:file = slhaea" for the SLHAea interface.
126 pythiaOptions.push_back("SLHA:file = slhaea");
127
128 // If the collider energy is not given in the list of Pythia options, we set it to 13 TeV by default.
129 // We search for the substring "Beams:e", meaning that if any of the Pythia options "Beams:eCM", "Beams:eA"
130 // or "Beams:eB" are present we don't apply the default.
131 bool has_beam_energy_option = std::any_of(pythiaOptions.begin(), pythiaOptions.end(), [](const str& s){ return s.find("Beams:e") != str::npos; });
132 if (!has_beam_energy_option)
133 {
134 pythiaOptions.push_back("Beams:eCM = 13000");
135 logger() << LogTags::debug << "Could not find a beam energy in the list of Pythia settings. Will add the setting 'Beams:eCM = 13000'." << EOM;
136 }
137
138 // Variables needed for the xsec veto
139 std::stringstream processLevelOutput;
140 str _junk, readline;
141 int code, nxsec;
142 double xsec, totalxsec;
143
144 // Each thread needs an independent Pythia instance at the start
145 // of each event generation loop.
146 // Thus, the actual Pythia initialization is
147 // *after* COLLIDER_INIT, within omp parallel.
148
149 result.clear();
150
151 // Add the thread-specific seed to the Pythia options
152 str seed = std::to_string(int(Random::draw() * 899990000.));
153 pythiaOptions.push_back("Random:seed = " + seed);
154
155 #ifdef COLLIDERBIT_DEBUG
156 cout << DEBUG_PREFIX << "getPythia"+model_suffix+": My Pythia seed is: " << seed << endl;
157 #endif
158
159 try
160 {
161 result.init(pythia_doc_path, pythiaOptions, &slha, processLevelOutput);
162 }
163 catch (typename Py8Collider<PythiaT,EventT,hepmc_writerT>::InitializationError& e)
164 {
165 // Append new seed to override the previous one
166 int newSeedBase = int(Random::draw() * 899990000.);
167 pythiaOptions.push_back("Random:seed = " + std::to_string(newSeedBase));
168 try
169 {
170 result.init(pythia_doc_path, pythiaOptions, &slha, processLevelOutput);
171 }
172 catch (typename Py8Collider<PythiaT,EventT,hepmc_writerT>::InitializationError& e)
173 {
174 #ifdef COLLIDERBIT_DEBUG
175 cout << DEBUG_PREFIX << "Py8Collider::InitializationError caught in getPy8Collider. Will discard this point." << endl;
176 #endif
177 piped_invalid_point.request("Bad point: Pythia can't initialize");
178 wrapup();
179 return;
180 }
181 }
182
183 // Should we apply the xsec veto and skip event generation?
184
185 // - Get the upper limt xsec as estimated by Pythia
186 code = -1;
187 nxsec = 0;
188 totalxsec = 0.;
189 while(true)
190 {
191 std::getline(processLevelOutput, readline);
192 std::istringstream issPtr(readline);
193 issPtr.seekg(47, issPtr.beg);
194 issPtr >> code;
195 if (!issPtr.good() && nxsec > 0) break;
196 issPtr >> _junk >> xsec;
197 if (issPtr.good())
198 {
199 totalxsec += xsec;
200 nxsec++;
201 }
202 }
203
204 #ifdef COLLIDERBIT_DEBUG
205 cout << DEBUG_PREFIX << "totalxsec [fb] = " << totalxsec * 1e12 << ", veto limit [fb] = " << xsec_veto_fb << endl;
206 #endif
207
208 // - Check for NaN xsec
209 if (Utils::isnan(totalxsec))
210 {
211 #ifdef COLLIDERBIT_DEBUG
212 cout << DEBUG_PREFIX << "Got NaN cross-section estimate from Pythia." << endl;
213 #endif
214 piped_invalid_point.request("Got NaN cross-section estimate from Pythia.");
215 wrapup();
216 return;
217 }
218
219 // - Wrap up loop if veto applies
220 if (totalxsec * 1e12 < xsec_veto_fb)
221 {
222 #ifdef COLLIDERBIT_DEBUG
223 cout << DEBUG_PREFIX << "Cross-section veto applies. Will now call Loop::wrapup() to skip event generation for this collider." << endl;
224 #endif
225 wrapup();
226 } else {
227
228 // Create a dummy event to make Pythia fill its internal list of process codes
229 EventT dummy_pythia_event;
230 try
231 {
232 result.nextEvent(dummy_pythia_event);
233 }
234 catch (typename Py8Collider<PythiaT,EventT,hepmc_writerT>::EventGenerationError& e)
235 {
236 // Try again...
237 try
238 {
239 result.nextEvent(dummy_pythia_event);
240 }
241 catch (typename Py8Collider<PythiaT,EventT,hepmc_writerT>::EventGenerationError& e)
242 {
243 piped_invalid_point.request("Failed to generate dummy test event. Will invalidate point.");
244

247 #endif

265 #define GET_SPECTRUM_AND_DECAYS_FOR_PYTHIA_SUSY(NAME, SPECTRUM) \
266 void NAME(SLHAstruct& result) \
267 { \
268 using namespace Pipes::NAME; \
269 static const int slha_version = runOptions->getValueOrDef<int>(2, "slha_version"); \
270 static const bool write_summary_to_log = \
271 runOptions->getValueOrDef<bool>(false, "write_summary_to_log"); \
272 \
273 if ((slha_version != 1) && (slha_version != 2)) \
274 { \
275 ColliderBit_error().raise(LOCAL_INFO, \
276 "The option 'slha_version' must be set to 1 or 2 (default)."); \
277 } \
278 result.clear(); \
279 /* Get decays */ \
280 result = Dep::decay_rates->getSLHAea(slha_version, false, *Dep::SLHA_pseudonyms); \
281 /* Get spectrum */ \
282 SLHAstruct slha_spectrum = Dep::SPECTRUM->getSLHAea(slha_version); \
283 result.insert(result.begin(), slha_spectrum.begin(), slha_spectrum.end()); \
284 /* Add MODSEL block if not found */ \
285 if(result.find("MODSEL") == result.end()) \
286 { \
287 SLHAea::Block block("MODSEL"); \
288 block.push_back("BLOCK MODSEL # Model selection"); \
289 SLHAea::Line line; \
290 line << 1 << 0 << "# Tell Pythia that this is a SUSY model."; \
291 block.push_back(line); \
292 result.push_front(block); \
293 } \
294 \
295 if (write_summary_to_log) \
296 { \
297 std::stringstream SLHA_log_output; \
298 SLHA_log_output << "SLHA" << slha_version << " input to Pythia:\n" << result.str() \
299 << "\n"; \
300 logger() << SLHA_log_output.str() << EOM; \
301 } \
