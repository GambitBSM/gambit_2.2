113 inline bool get_unified_fromHadron(const HepMC3::GenParticlePtr &gp, const std::vector<HepMC3::GenParticlePtr> &pevt, int i)
114 {
115 // This function mimics exactly what the Py8Utils.cpp function does, but for HepMC3 events.
116 // This seems highly unlikely to change - apparently this is just the standard way it's done.
117 // Note that the meaningless int argument is to make sure that the same function call works both for HepMC3
118 // and Pythia.
119 if (MCUtils::PID::isHadron(gp->pid())) return true;
120 if (HEPMC3_isParton(abs(gp->pid()))) return false; // stop the walking at the end of the hadron level
121 auto parent_vector = (gp->parents());
122 if (parent_vector.size() == 0) return false;
123 for (const HepMC3::GenParticlePtr& parent : parent_vector)
124 {
125 if (get_unified_fromHadron(parent, pevt, i)) return true;
126 }
127 return false;
128 }
