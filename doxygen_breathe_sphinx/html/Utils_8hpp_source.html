215 inline std::map<const HEPUtils::Jet*,bool> generateBTagsMap(const std::vector<const HEPUtils::Jet*>& jets,
216 double bTagEff, double cMissTagEff, double otherMissTagEff,
217 double pTmin = 0., double absEtaMax = DBL_MAX)
218 {
219 std::map<const HEPUtils::Jet*,bool> bTagsMap;
220 for (const HEPUtils::Jet* j : jets)
221 {
222 bool genBTag = false;
223 if((j->pT() > pTmin) && (j->abseta() < absEtaMax))
224 {
225 if(j->btag())
226 {
227 if(random_bool(bTagEff)) { genBTag = true; }
228 }
229 else if(j->ctag())
230 {
231 if(random_bool(cMissTagEff)) { genBTag = true; }
232 }
233 else
234 {
235 if(random_bool(otherMissTagEff)) { genBTag = true; }
236 }
237 }
238 bTagsMap[j] = genBTag;
239 }
240 return bTagsMap;
241 }

245 inline size_t binIndex(NUM1 val, const std::vector<NUM2>& binedges, bool allow_overflow=false) {
246 if (val < binedges.front()) return -1;

248 return std::distance(binedges.begin(), --std::upper_bound(binedges.begin(), binedges.end(), val));

267 template <typename MOM>
268 inline std::vector<std::shared_ptr<HEPUtils::Jet>> get_jets(const std::vector<MOM*>& moms, double R,
269 double ptmin=0*GeV, FJNS::JetAlgorithm alg=FJNS::antikt_algorithm) {
270 // Make PseudoJets
271 std::vector<FJNS::PseudoJet> constituents;
272 for (const MOM* p : moms) constituents.push_back(HEPUtils::mk_pseudojet(*p));
273 // Run clustering
274 std::vector<FJNS::PseudoJet> jets = HEPUtils::get_jets(constituents, R, ptmin, alg);
275 // Make newly-allocated Jets
276 std::vector<std::shared_ptr<HEPUtils::Jet>> rtn;
277 for (const FJNS::PseudoJet& j : jets) rtn.push_back(std::make_shared<HEPUtils::Jet>(HEPUtils::mk_p4(j)));
278 return rtn;
279 }

283 inline bool object_in_cone(const HEPUtils::Event& e, const HEPUtils::P4& p4, double ptmin, double rmax, double rmin=0.05) {
284 for (const HEPUtils::Particle* p : e.visible_particles())
285 if (p->pT() > ptmin && HEPUtils::in_range(HEPUtils::deltaR_eta(p4, *p), rmin, rmax)) return true;
286 for (const HEPUtils::Jet* j : e.jets())
287 if (j->pT() > ptmin && HEPUtils::in_range(HEPUtils::deltaR_eta(p4, *j), rmin, rmax)) return true;
288 return false;
289 }

298 void removeOverlap(MOMPTRS1& momstofilter, const MOMPTRS2& momstocmp, double deltaRMax, bool use_rapidity=false, double pTmax = DBL_MAX, double btageff = 0)
299 {
300 ifilter_reject(momstofilter, [&](const typename MOMPTRS1::value_type& mom1)
301 {
302 for (const typename MOMPTRS2::value_type& mom2 : momstocmp) {

304 if (dR < deltaRMax && mom1->pT() < pTmax && ( !amIaBJet(mom1) || !random_bool(btageff) ) ) return true;
305 }
306 return false;
