113 inline void smearElectronEnergy(std::vector<HEPUtils::Particle*>& electrons) {
114
115 // Now loop over the electrons and smear the 4-vectors
116 for (HEPUtils::Particle* e : electrons) {
117
118 // Calculate resolution
119 // for pT > 0.1 GeV, E resolution = |eta| < 0.5 -> sqrt(0.06^2 + pt^2 * 1.3e-3^2)
120 // |eta| < 1.5 -> sqrt(0.10^2 + pt^2 * 1.7e-3^2)
121 // |eta| < 2.5 -> sqrt(0.25^2 + pt^2 * 3.1e-3^2)
122 double resolution = 0;
123 const double abseta = e->abseta();
124 if (e->pT() > 0.1 && abseta < 2.5) {
125 if (abseta < 0.5) {
126 resolution = HEPUtils::add_quad(0.06, 1.3e-3 * e->pT());
127 } else if (abseta < 1.5) {
128 resolution = HEPUtils::add_quad(0.10, 1.7e-3 * e->pT());
129 } else { // still |eta| < 2.5
130 resolution = HEPUtils::add_quad(0.25, 3.1e-3 * e->pT());
131 }
132 }
133
134 // Smear by a Gaussian centered on the current energy, with width given by the resolution
135 if (resolution > 0) {
136 std::normal_distribution<> d(e->E(), resolution);
137 double smeared_E = d(Random::rng());
138 if (smeared_E < e->mass()) smeared_E = 1.01*e->mass();
139 // double smeared_pt = smeared_E/cosh(e->eta()); ///< @todo Should be cosh(|eta|)?
140 e->set_mom(HEPUtils::P4::mkEtaPhiME(e->eta(), e->phi(), e->mass(), smeared_E));
141 }
142 }
143 }

150 inline void smearMuonMomentum(std::vector<HEPUtils::Particle*>& muons) {
151
152 // Now loop over the muons and smear the 4-vectors
153 for (HEPUtils::Particle* p : muons) {
154
155 // Calculate resolution
156 // for pT > 0.1 GeV, mom resolution = |eta| < 0.5 -> sqrt(0.01^2 + pt^2 * 2.0e-4^2)
157 // |eta| < 1.5 -> sqrt(0.02^2 + pt^2 * 3.0e-4^2)
158 // |eta| < 2.5 -> sqrt(0.05^2 + pt^2 * 2.6e-4^2)
159 double resolution = 0;
160 const double abseta = p->abseta();
161 if (p->pT() > 0.1 && abseta < 2.5) {
162 if (abseta < 0.5) {
163 resolution = HEPUtils::add_quad(0.01, 2.0e-4 * p->pT());
164 } else if (abseta < 1.5) {
165 resolution = HEPUtils::add_quad(0.02, 3.0e-4 * p->pT());
166 } else { // still |eta| < 2.5... but isn't CMS' mu acceptance < 2.4?
167 resolution = HEPUtils::add_quad(0.05, 2.6e-4 * p->pT());
168 }
169 }
170
171 // Smear by a Gaussian centered on the current pT, with width given by the resolution
172 std::normal_distribution<> d(p->pT(), resolution*p->pT());
173 double smeared_pt = d(Random::rng());
174 if (smeared_pt < 0) smeared_pt = 0;
175 // const double smeared_E = smeared_pt*cosh(mu->eta()); ///< @todo Should be cosh(|eta|)?
176 // std::cout << "Muon pt " << mu_pt << " smeared " << smeared_pt << std::endl;
177 p->set_mom(HEPUtils::P4::mkEtaPhiMPt(p->eta(), p->phi(), p->mass(), smeared_pt));
178 }
179 }

188 inline void smearJets(std::vector<HEPUtils::Jet*>& jets) {
189
190 // Const resolution for now
195 //const double resolution = 0.03;
196
197 // Matthias jet smearing implemented roughly from https://arxiv.org/pdf/1607.03663.pdf
198 // Parameterisation can be still improved as functional form is given
199 // Pileup of <mu>=25 is taken, as JER depends strongly on mu
200 // CMS does not include information about JER at eta>1.3
201 const std::vector<double> binedges_eta = {0,10.};
202 const std::vector<double> binedges_pt = {0,20,30,40,50.,70.,100.,150.,200.,1000.,10000.};
203 const std::vector<double> JetsJER = {0.3,0.2,0.16,0.145,0.12,0.1,0.09,0.08,0.06,0.05};
204 static HEPUtils::BinnedFn2D<double> _resJets2D(binedges_eta,binedges_pt,JetsJER);
205
206 // Now loop over the jets and smear the 4-vectors
207 for (HEPUtils::Jet* jet : jets) {
208 const double resolution = _resJets2D.get_at(jet->abseta(), jet->pT());
209 std::normal_distribution<> d(1., resolution);
210 // Smear by a Gaussian centered on 1 with width given by the (fractional) resolution
211 double smear_factor = d(Random::rng());
212 jet->set_mom(HEPUtils::P4::mkXYZM(jet->mom().px()*smear_factor, jet->mom().py()*smear_factor, jet->mom().pz()*smear_factor, jet->mass()));
213 }
214 }

241 if (bjets.empty()) return;

243 const static std::vector<double> binedges_et = {25., 40., 60., 80., 100., 150., 200., 250., 300., 400., 500.,DBL_MAX };
244 const static std::vector<double> bineffs_et = {0.58, 0.61, 0.63, 0.64, 0.65, 0.62,0.6, 0.58, 0.56, 0.52, 0.48};
